void leftRotate(int arr[], int n, int d)
{
    d = d%n;

    int temp[d];
    for(inr i = 0; i<d; i++)
    {
        temp[i] = arr[i];
    }

    for (int i = d; i<n; i++)
    {
        arr[i-d]= arr[i];
    }

    for(int i = n-d; i<n; i++)

}
