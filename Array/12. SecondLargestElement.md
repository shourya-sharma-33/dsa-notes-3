optimal is again looping and comparing
and if slargest variable is greater than element and less than the largest than we can say its the second largest and the same logic follows

```cpp
int secondLargest(vector<int> &a, int n)
{
    int largest = a[0];
    int sLar = -1;
    for (int i = 1; i<n; i++)
    {
        if(a[i]> largest)
        {
            sLar = largest;
            largest = a[i];
        }
        else if(a[i]<largest && a[i]> sLar)
        {
            slargest = a[i];
        }
    }
    return sLar;
}
```
