we can declare a temporary int as temp = arr[0] and we can loop in the array and we can declare arr[i-1] = arr[i] and say arr[n-1] = temp;

```cpp
vector<int> rotateArray(vector<int> &arr, int n)
{
    int temp = arr[0];
    for (int i = 1;  i<n; i++)
    {
        arr[i-1] = arr[i];
    }
    arr[n-1]=temp;
    return arr;
}
```
